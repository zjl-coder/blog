# 求出和最大的连续子数组的下标
```js
function getMaxSumArr(arr){
  // 假设数组的第一个元素不需要与其他元素相加即是和的最大值
  var maxStart = 0;   // 则开始的下标为0
  var maxEnd = 0;     // 则结束下标为0
  var maxSum = arr[1];  // 最大和为第一个元素

  var idx = 0;  // 游标，标记有可能成为开始下标的一个标量

  var sum = 0;  // 所求的和

  // 对数组进行循环
  for(var i = 0; i < arr.length; i++){
    // 获取每个元素与之前的元素和进行相加
    sum = sum + arr[i];
    // 判断是否大于原来的最大值
    if(sum > maxSum){
      // 大于，则设置新的最大值
      maxSum = sum;
      // 并将结束下标标记在该元素下，表示到目前为止可以得到的最大值的下标
      maxEnd = i;
    }

    // 额外说明，如果下标不

    // 结束下标如果比游标大或等于，则表示到目前为止，游标到结束下标之间的数组元素和是最大值，
    // 则将开始下标与游标进行同步，否则，游标比结束下标大，则说明到目前为止，没有超过之前的最大值，开始和结束的下标都没有变化
    if(maxEnd >= idx){
      maxStart = idx;
    }

    // 如果所加之和出现小于0的情况，则说明后续相加的元素都会比不加之前的sum的值更大。
    // 标明该元素是个负数
    // 那么将sum的和置零，表示从新开始累加，如果一直是负数，则一直重置零
    // 将游标标向下一个元素，如果一直是负数，则一直往下标，（虽然一直往下标，但是开始和结束的下标不一定会跟着变化）
    if(sum < 0){
      sum = 0;
      idx = i+1;
    }
    
  }
  return [maxStart, maxEnd, maxSum];
}

var arr1 = [31,-41,59,26,-53,58,97,-93,-23,84];
var res = getMaxSumArr(arr1);
console.log(res);
```
